# MixCut Windows 对齐 macOS v0.3.0 / v0.3.1 / v0.3.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 Windows 版 v0.5.0 基础上，补齐 Mac v0.3.0 / v0.3.1 / v0.3.2 三个版本的全部业务功能 —— 手动分镜编辑器（场景 A + 场景 B + AI 反推）、9:16 全局视频比例、筛选导出、分镜首帧缩略图、UpdateChecker + 顶部 banner、自定义组合保持勾选顺序。

**Architecture:**
- 数据模型新增 2 字段（`MixStrategy.IsCustomGroup` / `MixScheme.IsManuallyEdited`），自定义方案挂在一个真实的「自定义组合」`MixStrategy` 容器下；
- WPF 端用 hover Trigger + ItemsControl + ContentPresenter 实现 SwiftUI 行间 ⊕ 与右侧抽屉的等价交互，**不强行 1:1 复刻 SwiftUI 控件**；
- `UpdateChecker` 用 `HttpClient` + 5s 超时双轨容灾（GitHub → Gitee），URL 直接用探测成功那条路。

**Tech Stack:** C# 12 + WPF + .NET 8（`net8.0-windows`），CommunityToolkit.Mvvm 8.4，EF Core 8 + SQLite，Serilog。

```csharp
    /// <summary>
    /// 为所有老项目补建「自定义组合」容器策略（首次升级后跑一次）。
    /// 对齐 Mac v0.3.0 ensureCustomGroupStrategy。
    /// </summary>
    private static void EnsureCustomGroupStrategy(
        IDbContextFactory<MixCutDbContext> dbFactory,
        AppSettings settings,
        ILogger logger)
    {
        if (settings.DidEnsureCustomGroupStrategyV1)
        {
            return;
        }

        try
        {
            using var db = dbFactory.CreateDbContext();
            var projects = db.Projects.Include(p => p.Strategies).ToList();
            var added = 0;
            foreach (var p in projects)
            {
                if (p.Strategies.Any(s => s.IsCustomGroup))
                {
                    continue;
                }
                db.Strategies.Add(new MixStrategy
                {
                    Name = "自定义组合",
                    Style = string.Empty,
                    StrategyDescription = "手动挑选分镜组合的方案",
                    TargetAudience = string.Empty,
                    NarrativeStructure = string.Empty,
                    TargetDuration = 0,
                    IsCustomGroup = true,
                    ProjectId = p.Id,
                });
                added++;
            }

            if (added > 0)
            {
                db.SaveChanges();
                logger.LogInformation("[CustomGroupMigration] 已为 {Count} 个老项目补建「自定义组合」策略", added);
            }
            else
            {
                logger.LogInformation("[CustomGroupMigration] 无需补建（已有 / 无老项目）");
            }

            settings.DidEnsureCustomGroupStrategyV1 = true;
        }
        catch (Exception ex)
        {
            logger.LogError(ex, "[CustomGroupMigration] 迁移失败");
        }
    }
```

    private static bool SchemeContainsSegment(MixScheme scheme, Segment segment) =>

    private static void RenumberPositions(MixScheme scheme)
    {

            ordered[i].Position = i + 1;
        }
    }

### Task 8: InsertSegment 插入方法

**Files:**
- Modify: `src/MixCut/ViewModels/SchemeViewModel.cs`

- [ ] **Step 1: 在 `RemoveSegment` 上方加 InsertSegment**

```csharp
    /// <summary>
    /// 在方案的指定位置插入一个新分镜（position 从 1 开始；position=count+1 表示尾部追加）。
    /// 返回 false 表示已包含该分镜（拒绝插入）；true 表示成功。
    /// </summary>
    public bool InsertSegment(Segment segment, int position, MixScheme scheme)
    {
        if (_context is null)
        {
            return false;
        }

        if (SchemeContainsSegment(scheme, segment))
        {
            return false;
        }

        // 拿当前 db 上下文中追踪的 scheme（避免插入断开实体的导航）
        var trackedScheme = _context.Schemes
            .Include(s => s.SchemeSegments)
            .FirstOrDefault(s => s.Id == scheme.Id);
        if (trackedScheme is null)
        {
            return false;
        }

        var trackedSegment = _context.Segments.FirstOrDefault(s => s.Id == segment.Id)
            ?? throw new InvalidOperationException($"Segment {segment.Id} 不在 db 上下文中");

        var clampedPos = Math.Clamp(position, 1, trackedScheme.SchemeSegments.Count + 1);

        // 后续位置整体后移
        foreach (var ss in trackedScheme.SchemeSegments)
        {
            if (ss.Position >= clampedPos)
            {
                ss.Position++;
            }
        }

        _context.SchemeSegments.Add(new SchemeSegment
        {
            Position = clampedPos,
            Reasoning = string.Empty,
            PositionReasoning = string.Empty,
            Scheme = trackedScheme,
            Segment = trackedSegment,
        });

        MarkAsEdited(trackedScheme);
        SaveContext();

        // 内存集合同步刷新
        var existing = Strategies.SelectMany(st => st.Schemes).FirstOrDefault(s => s.Id == scheme.Id);
        if (existing is not null && existing != trackedScheme)
        {
            // 内存对象不一致，重新加载该项目
            // 由 caller 决定要不要 reload；这里仅 db 已落
        }
        return true;
    }
```

### Task 9: ReplaceSegment 替换方法

**Files:**
- Modify: `src/MixCut/ViewModels/SchemeViewModel.cs`

- [ ] **Step 1: 在 InsertSegment 下方加 ReplaceSegment**

```csharp
    /// <summary>
    /// 把指定 SchemeSegment 替换为新分镜。
    /// 返回 false 表示新分镜已在方案中（拒绝替换，避免重复）；true 表示成功。
    /// </summary>
    public bool ReplaceSegment(SchemeSegment schemeSeg, Segment newSegment, MixScheme scheme)
    {
        if (_context is null)
        {
            return false;
        }

        // 同一分镜（含被替换自身）不算重复
        if (scheme.SchemeSegments.Any(ss => ss.SegmentId == newSegment.Id && ss.Id != schemeSeg.Id))
        {
            return false;
        }

        var tracked = _context.SchemeSegments.FirstOrDefault(ss => ss.Id == schemeSeg.Id);
        if (tracked is null)
        {
            return false;
        }

```csharp
    // ---- v0.3.0：自定义方案 AI 反推 ----

    /// <summary>
    /// 根据用户手动选择的分镜反推方案元信息（name / narrativeStructure / targetAudience / schemeDescription / style）。
    /// 失败返回 null，让上层落默认名 + banner 提示，不抛错阻断流程。
    /// </summary>
    public async Task<CustomSchemeMetadata?> InferMetadataAsync(
        IReadOnlyList<Segment> orderedSegments,
        CancellationToken cancellationToken = default)
    {
        if (orderedSegments.Count == 0)
        {
            return null;
        }

        try
        {
            var template = _promptLoader.LoadPrompt("custom_scheme_inference")
                ?? throw new InvalidOperationException("custom_scheme_inference prompt 未加载");

            var segmentsData = orderedSegments.Select((seg, idx) => new
            {
                position = idx + 1,
                semanticTypes = seg.SemanticTypes.Select(t => t.ToLabel()).ToArray(),
                text = seg.Text ?? string.Empty,
                duration = Math.Round(seg.Duration, 1),
            });
            var segmentsJson = System.Text.Json.JsonSerializer.Serialize(
                segmentsData,
                new System.Text.Json.JsonSerializerOptions { WriteIndented = false });
            var prompt = template.Replace("{{SEGMENTS_JSON}}", segmentsJson);

            var provider = _providerManager.GetActiveProvider();
            var raw = await provider.GenerateAsync(prompt, cancellationToken);

            // 抠 JSON：复用现有 JsonTruncationRepair 或简单 fence 剥离
            var jsonText = ExtractJsonObject(raw) ?? raw;
            var meta = System.Text.Json.JsonSerializer.Deserialize<CustomSchemeMetadata>(jsonText);
            _logger.LogInformation("[CustomSchemeInference] 反推完成: name={Name}", meta?.Name);
            return meta;
        }
        catch (Exception ex)
        {
            _logger.LogWarning(ex, "[CustomSchemeInference] 反推失败，落默认元信息");
            return null;
        }
    }

    private static string? ExtractJsonObject(string raw)
    {
        if (string.IsNullOrWhiteSpace(raw))
        {
            return null;
        }
        var s = raw.Trim();
        // 去 markdown fence
        if (s.StartsWith("```"))
        {
            var firstNewline = s.IndexOf('\n');
            if (firstNewline > 0) s = s[(firstNewline + 1)..];
            var fenceEnd = s.LastIndexOf("```", StringComparison.Ordinal);
            if (fenceEnd > 0) s = s[..fenceEnd];
            s = s.Trim();
        }
        var start = s.IndexOf('{');
        var end = s.LastIndexOf('}');
        if (start < 0 || end <= start) return null;
        return s.Substring(start, end - start + 1);
    }
```

    // ---- v0.3.0：自定义方案创建 ----

    /// <summary>
    /// 用户在分镜库勾选若干分镜并保持顺序后调用：
    /// 1) 在「自定义组合」策略下创建占位 scheme（默认名「自定义 #N」）
    /// 2) 异步 AI 反推元信息覆盖默认名（失败兜底）
    /// 3) 返回最终的 MixScheme（供调用方跳转选中）
    /// </summary>
    public async Task<MixScheme?> CreateCustomSchemeAsync(
        IReadOnlyList<Segment> orderedSegments,
        Project project,
        CancellationToken cancellationToken = default)
    {
        if (_context is null)
        {
            return null;
        }
        if (orderedSegments.Count < 2)
        {
            return null;
        }

        // 找/建自定义组合策略
        var customGroup = _context.Strategies
            .Include(s => s.Schemes)
            .FirstOrDefault(s => s.ProjectId == project.Id && s.IsCustomGroup);
        if (customGroup is null)
        {
            customGroup = new MixStrategy
            {
                Name = "自定义组合",
                IsCustomGroup = true,
                ProjectId = project.Id,
                StrategyDescription = "手动挑选分镜组合的方案",
            };
            _context.Strategies.Add(customGroup);
            _context.SaveChanges();
        }

        var nextIdx = (customGroup.Schemes.Count == 0 ? 0 : customGroup.Schemes.Max(s => s.VariationIndex)) + 1;
        var placeholder = $"自定义 #{nextIdx}";
        var totalDuration = orderedSegments.Sum(s => s.Duration);

        var scheme = new MixScheme
        {
            VariationIndex = nextIdx,
            SchemeIndex = $"custom_{nextIdx:D3}",
            Name = placeholder,
            Style = string.Empty,
            SchemeDescription = string.Empty,
            TargetAudience = string.Empty,
            NarrativeStructure = string.Empty,
            EstimatedDuration = totalDuration,
            IsManuallyEdited = false, // 自定义方案不算 AI 被改
            StrategyId = customGroup.Id,
            ProjectId = project.Id,
        };
        _context.Schemes.Add(scheme);

        // 按用户给的顺序建 SchemeSegment
        for (var i = 0; i < orderedSegments.Count; i++)
        {
            var tracked = _context.Segments.FirstOrDefault(s => s.Id == orderedSegments[i].Id);
            if (tracked is null)
            {
                continue;
            }
            _context.SchemeSegments.Add(new SchemeSegment
            {
                Position = i + 1,
                Reasoning = string.Empty,
                PositionReasoning = string.Empty,
                Scheme = scheme,
                Segment = tracked,
            });
        }

        _context.SaveChanges();
        _logger.LogInformation("[CustomScheme] 占位方案已落库: name={Name} segments={Count}", placeholder, orderedSegments.Count);

        // 异步反推（失败不阻断）
        var meta = await _schemeService.InferMetadataAsync(orderedSegments, cancellationToken);
        if (meta is not null && !string.IsNullOrWhiteSpace(meta.Name))
        {
            scheme.Name = meta.Name!;
            scheme.NarrativeStructure = meta.NarrativeStructure ?? string.Empty;
            scheme.TargetAudience = meta.TargetAudience ?? string.Empty;
            scheme.SchemeDescription = meta.SchemeDescription ?? string.Empty;
            scheme.Style = meta.Style ?? string.Empty;
            _context.SaveChanges();
            _logger.LogInformation("[CustomScheme] 元信息反推完成: {Name}", scheme.Name);
        }

        // 刷新内存集合
        LoadSchemes(project);
        return Schemes.FirstOrDefault(s => s.Id == scheme.Id);
    }

using System.Collections.ObjectModel;
using System.Windows;
using System.Windows.Controls;
using MixCut.Models;
using MixCut.ViewModels.Cards;

namespace MixCut.Views.SegmentLibrary;

public partial class ArrangeOrderSheet : Window
{
    public ObservableCollection<SegmentPickerItem> OrderedSegments { get; }
    public string Hint => $"共 {OrderedSegments.Count} 个分镜 · 调整顺序后生成方案";
    public string GenerateButtonText => $"生成方案 ({OrderedSegments.Count})";

    public IReadOnlyList<Segment>? Result { get; private set; }

    public ArrangeOrderSheet(IReadOnlyList<Segment> initialOrder)
    {
        InitializeComponent();
        OrderedSegments = new ObservableCollection<SegmentPickerItem>(
            initialOrder.Select(s => new SegmentPickerItem(s, false)));
        DataContext = this;
    }

    private void OnMoveLeftClick(object sender, RoutedEventArgs e)
    {
        if (sender is Button btn && btn.Tag is SegmentPickerItem item)
        {
            var idx = OrderedSegments.IndexOf(item);
            if (idx > 0)
            {
                OrderedSegments.Move(idx, idx - 1);
            }
        }
    }

    private void OnMoveRightClick(object sender, RoutedEventArgs e)
    {
        if (sender is Button btn && btn.Tag is SegmentPickerItem item)
        {
            var idx = OrderedSegments.IndexOf(item);
            if (idx >= 0 && idx < OrderedSegments.Count - 1)
            {
                OrderedSegments.Move(idx, idx + 1);
            }
        }
    }

    private void OnCancelClick(object sender, RoutedEventArgs e)
    {
        DialogResult = false;
        Close();
    }

    private void OnGenerateClick(object sender, RoutedEventArgs e)
    {
        LoadingOverlay.Visibility = Visibility.Visible;
        Result = OrderedSegments.Select(x => x.Segment).ToList();
        DialogResult = true;
        Close();
    }
}
```

private async void OnCombineSchemeClick(object sender, RoutedEventArgs e)
{
    try
    {
        var selected = _vm.SelectedSegmentsInOrder;
        if (selected.Count < 2)
        {
            return;
        }
        var sheet = new ArrangeOrderSheet(selected) { Owner = Window.GetWindow(this) };
        if (sheet.ShowDialog() != true || sheet.Result is null)
        {
            return;
        }

        var schemeVm = App.AppHost.Services.GetRequiredService<SchemeViewModel>();
        var project = _vm.CurrentProject; // 需要 expose 出来
        if (project is null) return;

        var scheme = await schemeVm.CreateCustomSchemeAsync(sheet.Result, project);
        if (scheme is not null)
        {
            // 跳转到 SchemesView 并选中该方案
            var mainWin = Window.GetWindow(this) as MainWindow;
            mainWin?.NavigateToSchemesAndSelect(scheme);
            ToastService.Show($"已生成方案「{scheme.Name}」");
        }
        else
        {
            ToastService.Show("方案生成失败，请稍后重试");
        }
    }
    catch (Exception ex)
    {
        _logger.LogError(ex, "组合为方案失败");
        ToastService.Show($"组合失败: {ex.Message}");
    }
}

private void OnStrategyCheckClick(object sender, RoutedEventArgs e)
{
    if (sender is not CheckBox cb || cb.Tag is not MixStrategy strategy) return;
    var shouldSelect = cb.IsChecked == true;
    foreach (var scheme in strategy.Schemes)
    {
        if (shouldSelect) _selectedSchemeIds.Add(scheme.Id);
        else _selectedSchemeIds.Remove(scheme.Id);
    }
    RefreshExportListUI();
    UpdateExportButtonText();
}

private void OnSchemeCheckClick(object sender, RoutedEventArgs e)
{
    if (sender is not CheckBox cb || cb.Tag is not MixScheme scheme) return;

- [ ] **Step 1: 在 App.xaml.cs 加迁移方法**

```csharp
    /// <summary>
    /// 把已有分镜缩略图全部重生为「首帧」（Mac v0.3.1 对齐）。
    /// 用 AppSettings.DidRegenerateSegmentThumbnailsToFirstFrameV1 防重；并发不阻塞 UI。
    /// </summary>
    private static async Task RegenerateSegmentThumbnailsToFirstFrameAsync(
        IDbContextFactory<MixCutDbContext> dbFactory,
        FFmpegRunner ffmpeg,
        AppSettings settings,
        ILogger logger)
    {
        if (settings.DidRegenerateSegmentThumbnailsToFirstFrameV1)
        {
            return;
        }

        try
        {
            using var db = dbFactory.CreateDbContext();
            var segments = await db.Segments
                .Include(s => s.Video)
                .ToListAsync();

            var total = segments.Count;
            var regenerated = 0;

            var sem = new SemaphoreSlim(Math.Max(2, Environment.ProcessorCount / 2));
            var tasks = segments
                .Where(s => s.Video is not null && File.Exists(s.Video.LocalPath))
                .Select(async s =>
                {
                    await sem.WaitAsync();
                    try
                    {
                        if (!string.IsNullOrEmpty(s.ThumbnailPath) && File.Exists(s.ThumbnailPath))
                        {
                            try { File.Delete(s.ThumbnailPath); } catch { /* 忽略 */ }
                        }
                        var t = Math.Max(0, s.StartTime + 0.1);
                        var outDir = Path.Combine(AppPaths.Root, "Thumbnails");
                        Directory.CreateDirectory(outDir);
                        var outPath = Path.Combine(outDir, $"seg_{s.Id}.jpg");
                        await ffmpeg.GenerateThumbnailAsync(s.Video!.LocalPath, outPath, t);
                        s.ThumbnailPath = outPath;
                        Interlocked.Increment(ref regenerated);
                    }
                    catch (Exception ex)
                    {
                        logger.LogWarning(ex, "[ThumbMigration] 单分镜失败: id={Id}", s.Id);
                    }
                    finally
                    {
                        sem.Release();
                    }
                });
            await Task.WhenAll(tasks);

            await db.SaveChangesAsync();
            ThumbnailCache.Shared.Clear();
            logger.LogInformation("[ThumbMigration] 重生完成: {Done}/{Total}", regenerated, total);

            settings.DidRegenerateSegmentThumbnailsToFirstFrameV1 = true;
        }
        catch (Exception ex)
        {
            logger.LogError(ex, "[ThumbMigration] 迁移失败");
        }
    }

```csharp
using System.Net.Http;
using System.Net.Http.Headers;
using System.Text.Json;
using System.Text.Json.Serialization;
using Microsoft.Extensions.Logging;

namespace MixCut.Services.UpdateChecker;

/// <summary>
/// 版本检测服务（GitHub 优先 / Gitee 容灾）。对齐 macOS v0.3.1 UpdateChecker。
/// 设计意图：启动时探测当前网络能访问的源，ReleaseInfo.HtmlUrl 直接是用户能打开的页面。
/// </summary>
public sealed class UpdateChecker
{
    private const int TimeoutSec = 5;
    private const string GitHubUrl = "https://api.github.com/repos/RoshanGH/mixcut-windows/releases/latest";
    private const string GiteeUrl = "https://gitee.com/api/v5/repos/jinxiushanhehao/mixcut-windows/releases/latest";
    private const string GiteeTagPrefix = "https://gitee.com/jinxiushanhehao/mixcut-windows/releases/tag/";

    private readonly ILogger<UpdateChecker> _logger;

    public UpdateChecker(ILogger<UpdateChecker> logger)
    {
        _logger = logger;
    }

    public async Task<ReleaseInfo?> FetchLatestAsync(CancellationToken cancellationToken = default)
    {
        return await TryGitHubAsync(cancellationToken) ?? await TryGiteeAsync(cancellationToken);
    }

    private async Task<ReleaseInfo?> TryGitHubAsync(CancellationToken cancellationToken)
    {
        try
        {
            using var cts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken);
            cts.CancelAfter(TimeSpan.FromSeconds(TimeoutSec));
            using var http = new HttpClient();
            http.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue("application/vnd.github+json"));
            http.DefaultRequestHeaders.UserAgent.Add(new ProductInfoHeaderValue("MixCutWindows", "1.0"));
            var json = await http.GetStringAsync(GitHubUrl, cts.Token);
            var release = JsonSerializer.Deserialize<GitHubRelease>(json);
            if (release is null || string.IsNullOrEmpty(release.TagName))
            {
                return null;
            }
            _logger.LogInformation("[UpdateCheck] GitHub OK tag={Tag}", release.TagName);
            return new ReleaseInfo
            {
                Tag = release.TagName,
                Version = StripV(release.TagName),
                HtmlUrl = release.HtmlUrl,
                Source = "GitHub",
            };
        }
        catch (Exception ex)
        {
            _logger.LogInformation("[UpdateCheck] GitHub fail: {Msg}", ex.Message);
            return null;
        }
    }

    private async Task<ReleaseInfo?> TryGiteeAsync(CancellationToken cancellationToken)
    {
        try
        {
            using var cts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken);
            cts.CancelAfter(TimeSpan.FromSeconds(TimeoutSec));
            using var http = new HttpClient();
            var json = await http.GetStringAsync(GiteeUrl, cts.Token);
            var release = JsonSerializer.Deserialize<GiteeRelease>(json);
            if (release is null || string.IsNullOrEmpty(release.TagName))
            {
                return null;
            }
            _logger.LogInformation("[UpdateCheck] Gitee OK tag={Tag}", release.TagName);
            return new ReleaseInfo
            {
                Tag = release.TagName,
                Version = StripV(release.TagName),
                HtmlUrl = GiteeTagPrefix + release.TagName,
                Source = "Gitee",
            };
        }
        catch (Exception ex)
        {
            _logger.LogInformation("[UpdateCheck] Gitee fail: {Msg}", ex.Message);
            return null;
        }
    }

    public static bool IsNewer(string remote, string local)
    {
        if (string.IsNullOrWhiteSpace(remote) || string.IsNullOrWhiteSpace(local)) return false;
        int[] Parse(string v) => v.Split('.').Select(p => int.TryParse(p, out var n) ? n : 0).ToArray();
        var r = Parse(remote);
        var l = Parse(local);
        var max = Math.Max(r.Length, l.Length);
        for (var i = 0; i < max; i++)
        {
            var ri = i < r.Length ? r[i] : 0;
            var li = i < l.Length ? l[i] : 0;
            if (ri > li) return true;
            if (ri < li) return false;
        }
        return false;
    }

    private static string StripV(string tag) =>
        tag.StartsWith("v", StringComparison.OrdinalIgnoreCase) ? tag[1..] : tag;

    private sealed class GitHubRelease
    {
        [JsonPropertyName("tag_name")] public string TagName { get; set; } = string.Empty;
        [JsonPropertyName("html_url")] public string HtmlUrl { get; set; } = string.Empty;
    }
    private sealed class GiteeRelease
    {
        [JsonPropertyName("tag_name")] public string TagName { get; set; } = string.Empty;
    }
}

    {

        _settings = settings;
        _logger = logger;
        CurrentVersion = Assembly.GetExecutingAssembly().GetName().Version?.ToString(3) ?? "0.0.0";

    public async Task CheckSilentlyAsync()
    {
        try
        {
            var info = await _checker.FetchLatestAsync();
            if (info is null) return;
            if (!UpdateChecker.IsNewer(info.Version, CurrentVersion)) return;
            if (string.Equals(_settings.DismissedUpdateVersion, info.Version, StringComparison.Ordinal)) return;
            LatestInfo = info;
            HasUpdate = true;
            OnPropertyChanged(nameof(BannerText));
        }
        catch (Exception ex)

            _logger.LogWarning(ex, "[UpdateBanner] silent check 失败");
        }
    }

    public void DismissCurrent()
    {
        if (LatestInfo is null) return;
        _settings.DismissedUpdateVersion = LatestInfo.Version;
        HasUpdate = false;
    }

    public void OpenReleasePage()
    {
        if (LatestInfo is null) return;
        try
        {
            Process.Start(new ProcessStartInfo(LatestInfo.HtmlUrl) { UseShellExecute = true });
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "[UpdateBanner] 打开 release 页失败");
